Operator precedence


1. parenthesis ()
2. exponents
3. multiplication and division
4. addition and subtraction
let result = 1 + 2 * (3 + 4)
console.log(result) // result:15
let result = 1 + 2 * (7)
console.log(result) // result:15
let result = (1 + 2) * (7)
console.log(result) // result:21
