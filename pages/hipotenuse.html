How to find hipotenuse

let a
let b
let c

a = window.prompt("Enter side A") // 10
a = Number(a)

b = window.prompt("Enter side B") // 10
b = Number(b)

c = Math.sqrt(Math.pow(a, 2) + Math.pow(b, 2));
console.log("side C:" , c)

Javascript

let a
let b
let c

document.getElementById("submitButton").onclick = function(){

a = document.getElementById("aTextBox").value
a = Number(a)

b = Number(b)

c = Math.sqrt(Math.pow(a, 2) + Math.pow(b, 2))
