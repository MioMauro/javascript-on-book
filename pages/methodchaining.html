Method chaining


Method chaining =


calling one method after another, in one continuous line of code

let userName = "bro"
let letter = userName.charAt(0) // b
letter = letter.toUpperCase()
console.log(letter)

result: B (Uppercase)
let userName = "bro"
let letter = userName.charAt(0).toUpperCase() // same, but simplified
console.log(letter)

result: B (Uppercase)


let userName = "bro"
let letter = userName.charAt(0).toUpperCase().trim() // concatinating methods
console.log(letter)

result: B (Uppercase)
